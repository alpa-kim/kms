AsyncSession으로 엔티티/관계 일괄 UPSERT
-- 엔티티 UNWIND $entities AS e MERGE (n:Entity {name: e.name, kb_id: $kb_id}) ON CREATE SET n += e.properties ON MATCH SET n.description += '<SEP>' + e.description -- 관계 MERGE (from)-[rel:RELATED_TO]->(to) ON MATCH SET rel.weight += r.weight
